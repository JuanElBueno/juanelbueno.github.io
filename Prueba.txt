function abrirDocumento() {

    var urlDocumento = "https://example.com/tu_documento.docx";

    // Abre el documento en Word utilizando la URL
    window.open("https://view.officeapps.live.com/op/view.aspx?src=" + encodeURIComponent(urlDocumento), "_blank");
}
